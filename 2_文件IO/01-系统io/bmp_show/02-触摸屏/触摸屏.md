```c
输入事件
结构体：struct input_event     #include <linux/input.h>
cd /usr/include 
#include "input-event-codes.h" 包含了一些宏定义

struct input_event {
	struct timeval time;//输入事件发生的时间
	__u16 type;//事件的类型
		type == EV_KEY 按键
		type == EV_REL 鼠标
		type == EV_ABS 触摸事件
	__u16 code;//根据type的不同，code含义也会不一样
		type == EV_KEY 按键
			code 表示事件键值
				KEY_Q
				KEY_W
				...
				BTN_TOUCH	触摸点击
		type == EV_ABS 触摸事件
			code == ABS_X	x轴坐标
			code == ABS_Y	y轴坐标
            
	__s32 value;//根据type code的不同，value
	含义也会不一样
		type == EV_KEY 按键
			code == BTN_TOUCH
				value == 1 手指按下
				value == 0 手指松开
		type == EV_ABS 触摸事件
			code == ABS_X	x轴坐标
				value表示x轴的坐标值
			code == ABS_Y	y轴坐标
				value表示y轴的坐标值
		
};
```

```
//假设触摸屏打开的返回值为touch_fd

int read_x = -1,read_y = -1;//全局变量

int x = -1,y = -1;//局部变量 保存获取到的坐标
struct input_event ts;//保存触摸信息
while(1)
{
	read(touch_fd,&ts,sizeof(ts));
	
	if(ts.type == EV_ABS)//触摸屏时间
	{
		if(ts.code == ABS_X)//x轴
		{
			//获取x轴的坐标
			x = ts.value;
		}
		else if(ts.code == ABS_Y)//y轴
		{
			//获取y轴的坐标
			y = ts.value;			
		}
	}
	//什么时候获取到一个坐标
	if(x>=0 && y>=0)
	{
		printf("x = %d,y=%d\n",x,y);
		read_x = x;
		read_y = y;
		break;
	}	
}

//关闭触摸屏
close(touch_fd);

封装成 void get_xy();
```

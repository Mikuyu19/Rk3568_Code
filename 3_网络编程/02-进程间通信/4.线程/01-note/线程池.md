线程池是一种多线程处理方式，处理方法是将“生产者”线程提出的任务添加到“任务队列"，由之前创建的子线程自动取”任务队列“中选取任务执行

通常情况下，多线程变成都是在需要的时候，创建一个新的线程，然后完成指定的任务(线程函数)，完成后退出(就好像招临时工)，但是当我们需要创建大量的线程,并非执行一个任务后销毁,比如：

- 目录的拷贝

- web服务器的响应

- 数据库操作

    ...

那么像上面的这些情况，我们的程序需要面对大量的请求，但是这些任务又比较简单，那么程序就又能在不停的创建线程和销毁线程

线程池的作用：可以降低频繁创建和销毁线程带来的开销

线程池的技术思路：

> 一般采用预先创建线程的技术，也就是在应用启动之前创建好一定数量的线程(相当于招正式工，完成任务之后不会被解雇)
>
> 线程固定的执行一个“任务调度函数”，当“任务队列”没有任务，线程自动休眠(等待一个新的任务)，当任务到来的时候，被唤醒，再执行任务
>
> 线程完任务后不会被销毁，而是自动的执行任务队列的下一个任务
>
> 而且，当任务太多的时候，你可以写一个函数接口去增加线程的数量

// 任务结点
typedef struct task
{
    // 每一个任务结点保存一个任务的所有的信息，所谓的任务就是把一个文件拷贝到另外一个文件
    // 这个任务怎么保存
    // 任务的视线利用函数实现的，那么我们保存这个任务，只需要保存任务函数的任务就可以了
    // 还需要保存这个函数执行所需要的参数
    // 需要定义一个函数指针
    void *(*do_task)(void *arg);
    void *arg;         // 保存函数的参数
    struct task *next; // 指向下一个任务
} task;
typedef struct pthread_pool
{
    //"任务队列"是一种共享资源，所以需要互斥访问
    // 需要线程互斥锁
    pthread_mutex_t mutex;
    // 同时当任务队列中没有任务的时候，线程池的线程应该要休眠
    // 需要条件变量
    pthread_cond_t cond;
    //"任务队列"链表，指向第一个需要执行的任务
    // 所有的线程都需要从这个链表中取任务
    task *tasklist;
    // 指向线程id的数组，用来保存线程池中所有的线程id
    pthread_t *tids;
    // 线程池中正在执行的任务的线程数量
    unsigned int active_pthreads; // 保存开辟的线程的数量 
    //"任务队列"中当前任务的数量
    unsigned int cur_task_count;

    // 表示是否退出
    int shutdown; // 0表示退出 1表示不退出
} pthread_pool;

void add_task(pthread_pool *pool, void *(*fun_task)(void *arg), void *arg);

// 线程函数 文件内容拷贝
void *cp(void *arg)
{
    struct cp_file *p = (struct cp_file *)arg;
    printf("cpsrc:%s\n", p->src);
    printf("cpdest:%s\n", p->dest);
    int fd_src, fd_dest;
    // 只用读的方式打开源文件
    fd_src = open(p->src, O_RDONLY);
    // 只用写的方式打开目的文件
    fd_dest = open(p->dest, O_WRONLY | O_CREAT | O_TRUNC, 0777);

    // 循环读取源文件的数据写到目的文件
    while (1)
    {
        char buf[1024] = {0};
        int ret = read(fd_src, buf, sizeof(buf));
        if (ret == 0) // 读到文件末尾
            break;
        write(fd_dest, buf, ret);
    }

    close(fd_src);
    close(fd_dest);
    free(p);
}

// ./a.out /../abc/ccc  ../abc/bbb
void cp_dir(pthread_pool *pool, char *src_dir, char *dest_dir)
{
    /*打开源目录(从那个目录拷贝)*/
    DIR *dir = opendir(src_dir);

    /*分别获取源目录和目的目录绝对路径*/
    char srcdirname[1024] = {0};  // 保存原目录的绝对路径
    char destdirname[1024] = {0}; // 目的目录绝对路径
    char curdirname[1024] = {0};  // 保存当前工作目录的绝对路径

    getcwd(curdirname, sizeof(curdirname) - 1); // 获取当前工作目录的绝对路径

    chdir(src_dir);
    getcwd(srcdirname, sizeof(srcdirname) - 1); // 获取原目录的绝对路径
    chdir(curdirname);

    if (chdir(dest_dir) == -1)
    {
        // 不成功 创建目录
        mkdir(dest_dir, 0777);
        chdir(dest_dir);
    }
    getcwd(destdirname, sizeof(destdirname) - 1); // 获取原目录的绝对路径
    chdir(curdirname);

    printf("curdirname:%s\n", curdirname);
    printf("srcdirname:%s\n", srcdirname);
    printf("destdirname:%s\n", destdirname);
    /*让原目录成为目的目录的子目录名，获取这个子目录名的路径*/
    char *dirname = basename(srcdirname); // 摘路径

    char newdirname[2048] = {0};
    sprintf(newdirname, "%s/%s", destdirname, dirname);
    printf("newdirname:%s\n", newdirname);
    /*创建一个空目录*/
    int ret = mkdir(newdirname, 0777);
    if (ret == -1)
    {
        if (errno != EEXIST) // 出错
        {
            return;
        }
    }

    /*读取源目录的目录项*/
    struct dirent *dirp = NULL;
    while (dirp = readdir(dir))
    {
        // 先跳过两个隐藏目录
        if (strcmp(dirp->d_name, ".") == 0 || strcmp(dirp->d_name, "..") == 0)
            continue;

        // 将路径拼接
        char file_src[4096] = {0};
        sprintf(file_src, "%s/%s", srcdirname, dirp->d_name);
        // printf("file_src:%s\n", file_src);

        // 判断文件类型
        struct stat st;
        stat(file_src, &st);

        if (S_ISDIR(st.st_mode)) // 目录
        {
            cp_dir(pool, file_src, newdirname);
        }
        else if (S_ISREG(st.st_mode)) // 文件
        {
            // 拼接 newdirname/dirp->d_name
            char file_dest[4096] = {0};
            sprintf(file_dest, "%s/%s", newdirname, dirp->d_name);
            printf("file_src:%s\n", file_src);
            printf("file_dest:%s\n", file_dest);

            struct cp_file *cp_filename = malloc(sizeof(struct cp_file)); // 0x3000 路径变了
            strcpy(cp_filename->src, file_src);
            strcpy(cp_filename->dest, file_dest);

            // // 创建线程，让线程执行cp函数(拷贝文件内容)
            // pthread_create(&id[thread_count], NULL, cp, (void *)cp_filename);
            // 添加任务结点
            add_task(pool, cp, (void *)cp_filename);
            thread_count++;
        }
    }
}

/*任务调度函数(线程函数),让所有的线程都要执行这个函数，次函数会不断地从任务队列中取任务去执行*/
void *routine(void *arg)
{
    pthread_pool *pool = arg;
    while (1)
    {
        // p
        pthread_mutex_lock(&pool->mutex);
        while (pool->cur_task_count == 0 && pool->shutdown) // 当条件不满足(没有任务&&退出)
        {
            // 陷入休眠
            pthread_cond_wait(&pool->cond, &pool->mutex);
        }
        if (pool->cur_task_count == 0 && !pool->shutdown) // 退出
        {
            // v
            pthread_mutex_unlock(&pool->mutex);
            break;
        }
        // 取任务->把第一个任务结点从链表摘下来 头摘
        task *p = pool->tasklist->next;
        pool->tasklist->next = p->next;
        p->next = NULL;
        pool->cur_task_count--; // 任务数量-1
        pthread_mutex_unlock(&pool->mutex);
        // v

        // 解析任务结点->执行任务
        p->do_task(p->arg);

        // 释放结点
        free(p);
    }
}

void init_pool(pthread_pool *pool, unsigned int thread_num)
{
    // 初始化线程结构体
    // 初始化线程池互斥锁
    pthread_mutex_init(&pool->mutex, NULL);
    // 初始化线程池条件变量
    pthread_cond_init(&pool->cond, NULL);
    // 初始化链表
    pool->tasklist = malloc(sizeof(task));
    pool->tasklist->next = NULL;
    // 创建thread_num个线程，同时tids这个执行去记录所有线程id
    pool->tids = malloc(sizeof(pthread_t) * thread_num);

    // 保存线程数量
    pool->active_pthreads = thread_num;

    // 当前任务数量
    pool->cur_task_count = 0;

    // 不退出
    pool->shutdown = 1;

    // 开辟线程
    for (int i = 0; i < thread_num; i++)
    {
        pthread_create(&pool->tids[i], NULL, routine, (void *)pool);
    }
}

/*
    add_task:给任务队列增加任务，fun_task和arg两个成员封装到struct tack结点,把该结点加入到任务队列(尾插法)
*/
void add_task(pthread_pool *pool, void *(*fun_task)(void *arg), void *arg)
{
    // 需要把第二个参数和第三个参数封装成struct task的形式
    task *new_task = malloc(sizeof(task));
    new_task->do_task = fun_task;
    new_task->arg = arg;
    new_task->next = NULL;
    // p
    pthread_mutex_lock(&pool->mutex);
    // 把该结点加入到任务队列中去
    task *last = pool->tasklist;
    while (last->next)
    {
        last = last->next;
    }
    // 已经找到最后一个结点了
    last->next = new_task;
    pool->cur_task_count++; // 任务数量+1

    // 唤醒所有的线程
    pthread_cond_broadcast(&pool->cond);
    // v
    pthread_mutex_unlock(&pool->mutex);
}

// 销毁线程池
void destroy_pool(pthread_pool *pool)
{
    // 释放所有malloc空间，任务执行完毕->等待任务队里中结点数为0
    while (pool->cur_task_count != 0)
        ;

    pool->shutdown = 0;
    // 唤醒所有线程
    pthread_cond_broadcast(&pool->cond);

    // pthread_join
    for (int i = 0; i < pool->active_pthreads; i++)
    {
        pthread_join(pool->tids[i], NULL);
        printf("回收 tids[%d]\n", i);
    }

    free(pool->tasklist);
    free(pool->tids);
    free(pool);
}

int main(int argc, char **argv)
{
    if (argc != 3)
    {
        printf("Usage:%s <src_dir> <dest_dir>\n", argv[0]);
        return -1;
    }
    pthread_pool *pool = malloc(sizeof(pthread_pool));
    init_pool(pool, 10); // 初始化一个线程

    cp_dir(pool, argv[1], argv[2]);

    // 销毁线程池
    destroy_pool(pool);
}
